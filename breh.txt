$ProjectDir = "C:\Users\OpenClawAgent\Projects\ai-agent"
$PythonExe  = "$ProjectDir\.venv\Scripts\python.exe"
$ScriptPath = "$ProjectDir\run_pipeline.py"
$Username   = "OpenClawAgent"
$Password   = "YourPasswordHere123"

$Action = New-ScheduledTaskAction `
    -Execute $PythonExe `
    -Argument $ScriptPath `
    -WorkingDirectory $ProjectDir

$Trigger = New-ScheduledTaskTrigger -Daily -At "08:00"

$Settings = New-ScheduledTaskSettingsSet `
    -ExecutionTimeLimit (New-TimeSpan -Hours 1) `
    -RestartCount 2 `
    -RestartInterval (New-TimeSpan -Minutes 5) `
    -StartWhenAvailable

Register-ScheduledTask `
    -TaskName "OpenClawPipeline" `
    -TaskPath "\" `
    -Action $Action `
    -Trigger $Trigger `
    -Settings $Settings `
    -User $Username `
    -Password $Password `
    -RunLevel Highest `
    -Force




from dotenv import load_dotenv
from pathlib import Path

# Load .env using absolute path — required for Task Scheduler
load_dotenv(Path(__file__).parent / ".env")


Start-ScheduledTask -TaskName "OpenClawPipeline"
